async function handleCheckout() { const cart = getCart(); if (cart.length === 0) { alert("Cart is empty"); return; } const token = localStorage.getItem('token'); if (!token) { alert("Please login first"); window.location.href = 'auth.html'; return; } let total = 0; const bookIds = []; for (let i = 0; i < cart.length; i++) { total +=(cart[i].price || 0); bookIds.push(cart[i].id); } try { // First test if API is working const pingRes=await fetch('/api/ping'); const pingData=await pingRes.json(); console.log('API ping:', pingData); // Create order const orderRes=await fetch('/api/create-order', { method: 'POST' , headers: { 'Content-Type' : 'application/json' }, body: JSON.stringify({ amount: total, bookIds: bookIds }) }); if (!orderRes.ok) { const errorText=await orderRes.text(); console.error('Order creation failed:', errorText); throw new Error(`HTTP ${orderRes.status}: ${errorText.substring(0, 100)}`); } const order=await orderRes.json(); if (!order.id) throw new Error("Order creation failed"); const options={ key: 'rzp_test_SZWU44Pk0GG1I2' , amount: order.amount, currency: order.currency, name: 'Readera' , description: `Purchase of ${cart.length} book(s)`, order_id: order.id, handler: async function(response) { const verifyRes=await fetch('/api/verify-payment', { method: 'POST' , headers: { 'Content-Type' : 'application/json' }, body: JSON.stringify({ razorpay_order_id: response.razorpay_order_id, razorpay_payment_id: response.razorpay_payment_id, razorpay_signature: response.razorpay_signature, bookIds: bookIds }) }); const result=await verifyRes.json(); if (result.success) { let library=JSON.parse(localStorage.getItem('library') || '[]' ); for (let i=0; i < cart.length; i++) { if (!library.find(b=> b.id === cart[i].id)) { library.push(cart[i]); } } localStorage.setItem('library', JSON.stringify(library)); clearCart(); alert("✅ Payment successful! Books added to your library."); window.location.href = 'library.html'; } else { alert("❌ Payment verification failed: " + (result.error || 'Unknown error')); } }, prefill: { name: localStorage.getItem('userName') || 'Reader', email: localStorage.getItem('userEmail') || 'reader@example.com' }, theme: { color: '#8B4513' } }; const razorpay = new Razorpay(options); razorpay.open(); } catch (error) { console.error("Payment error:", error); alert("Payment failed: " + error.message); } }
